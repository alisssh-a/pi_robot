72 private void updateLogContent() {
73 StringBuilder content = new StringBuilder();
74 for (LogEntry entry : logSource.all()) {
75 if (entry.getLevel().level() >= currentFilterLevel.level()) {
76 content.append(String.format("[%s] %s%n",
77 entry.getLevel().name(), entry.getMessage()));
78 }
79 }
80 logContent.setText(content.toString());
81 logContent.setCaretPosition(logContent.getDocument().getLength());

84 private void saveLogToFile(ActionEvent e) {
85 JFileChooser fileChooser = new JFileChooser();
86 fileChooser.setDialogTitle(localizationManager.getString("save.log.title"));
87 fileChooser.setSelectedFile(new File("robot_log.txt"));
88
89 if (fileChooser.showSaveDialog(this) == JFileChooser.APPROVE_OPTION) {
90 File file = fileChooser.getSelectedFile();
91 try (PrintWriter writer = new PrintWriter(file)) {
92 for (LogEntry entry : logSource.all()) {
93 writer.printf("[%s] %s%n",
94 entry.getLevel().name(), entry.getMessage());
95 }
96 Logger.debug(localizationManager.getString("log.saved.message") + ": " + file.getAbsolutePath());
97 } catch (IOException ex) {
98 Logger.error(localizationManager.getString("log.save.error") + ": " + ex.getMessage());
99 JOptionPane.showMessageDialog(this,
100 localizationManager.getString("log.save.error.message"),
101 localizationManager.getString("error.title"),
