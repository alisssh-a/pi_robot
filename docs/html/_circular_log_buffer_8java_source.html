8// Кольцевой буфер для хранения логов с потокобезопасностью
9public class CircularLogBuffer implements Iterable<LogEntry> {
10 private final LogEntry[] buffer; // Массив для хранения записей
11 private final int capacity; // Максимальная ёмкость
12 private int head; // Индекс начала (где хранится самая старая запись)
13 private int size; // Текущее количество записей
14 private final ReentrantReadWriteLock lock = new ReentrantReadWriteLock(); // Для синхронизации
15
16 public CircularLogBuffer(int capacity) {
17 if (capacity <= 0) {
18 throw new IllegalArgumentException("Capacity must be positive");
19 }
20 this.capacity = capacity;
21 this.buffer = new LogEntry[capacity];
22 this.head = 0;
23 this.size = 0;
24 }
25
26 // Добавление записи (вытесняет старую, если буфер полон)
27 public void append(LogLevel logLevel, String message) {
28 LogEntry entry = new LogEntry(logLevel, message);
29 lock.writeLock().lock();
30 try {
31 int index = (head + size) % capacity;
32 buffer[index] = entry;
33 if (size < capacity) {
34 size++;
35 } else {
36 head = (head + 1) % capacity; // Сдвигаем начало, вытесняя старую запись
37 }
38 } finally {
39 lock.writeLock().unlock();
40 }
41 }
42
43 // Текущий размер буфера
44 public int size() {
45 lock.readLock().lock();
46 try {
47 return size;
48 } finally {
49 lock.readLock().unlock();
50 }
51 }
52
53 // Получение сегмента записей (от startFrom до startFrom + count)
54 public Iterable<LogEntry> range(int startFrom, int count) {
55 lock.readLock().lock();
56 try {
57 if (startFrom < 0 || startFrom >= size || count <= 0) {
58 return Collections.emptyList();
59 }
60 int endIndex = Math.min(startFrom + count, size);
61 int actualCount = endIndex - startFrom;
62 LogEntry[] result = new LogEntry[actualCount];
63 for (int i = 0; i < actualCount; i++) {
64 result[i] = buffer[(head + startFrom + i) % capacity];
65 }
66 // Возвращаем Iterable на основе массива
67 return () -> Arrays.stream(result).iterator();
68 } finally {
69 lock.readLock().unlock();
70 }
71 }
72
73 // Получение всех записей
74 public Iterable<LogEntry> all() {
75 return range(0, size());
76 }
77
78 // Потокобезопасный итератор (работает с копией данных)
79 @Override
80 public Iterator<LogEntry> iterator() {
81 return all().iterator();
82 }
