106 public void updateUI(Component component) {
107 if (component == null) return;
108
109 if (component instanceof JComponent) {
110 JComponent jComponent = (JComponent) component;
111 String translationKey = (String) jComponent.getClientProperty("translationKey");
112 if (translationKey != null) {
113 translationActions.entrySet().stream()
114 .filter(entry -> entry.getKey().isInstance(component))
115 .findFirst()
116 .ifPresent(entry -> entry.getValue().accept(component, getString(translationKey)));
117 }
118
119 if (component instanceof JMenu) {
120 for (Component menuChild : ((JMenu) component).getMenuComponents()) {
121 updateUI(menuChild);
122 }
123 }
124
125 for (Component child : jComponent.getComponents()) {
126 updateUI(child);
127 }
128 }
129
130 if (component instanceof JFrame) {
131 JFrame frame = (JFrame) component;
132 JRootPane rootPane = frame.getRootPane();
133 String titleKey = (String) rootPane.getClientProperty("translationKey");
134 if (titleKey != null) {
135 frame.setTitle(getString(titleKey));
136 }
137 updateUI(frame.getJMenuBar());
138 for (Window window : frame.getOwnedWindows()) {
139 updateUI(window);
140 }
141 }
