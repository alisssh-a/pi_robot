85 public void saveWindowState(JFrame frame) {
86 Properties props = loadProperties();
87 int state = frame.getExtendedState();
88 props.setProperty("main.state", String.valueOf(state));
89
90 if (state == Frame.NORMAL) {
91 int width = Math.max(frame.getWidth(), NORMAL_SIZE.width);
92 int height = Math.max(frame.getHeight(), NORMAL_SIZE.height);
93 props.setProperty("main.x", String.valueOf(frame.getX()));
94 props.setProperty("main.y", String.valueOf(frame.getY()));
95 props.setProperty("main.width", String.valueOf(width));
96 props.setProperty("main.height", String.valueOf(height));
97 Logger.debug(localizationManager.getString("saved.normal.state") + ": x=" + frame.getX() + ", y=" + frame.getY() + ", width=" + width + ", height=" + height);
98 } else {
99 Logger.debug(localizationManager.getString("saved.state") + ": state=" + state);
100 }
101
102 for (JInternalFrame internalFrame : desktopPane.getAllFrames()) {
103 String translationKey = (String) internalFrame.getClientProperty("translationKey");
104 if (translationKey != null) {
105 saveInternalFrameState(internalFrame, translationKey, props);
106 }
107 }

123 public void loadWindowState(JFrame frame) {
124 Properties props = loadProperties();
125 File configFile = new File(CONFIG_PATH);
126
127 frame.setMinimumSize(NORMAL_SIZE);
128
129 frame.addWindowStateListener(e -> {
130 int oldState = e.getOldState();
131 int newState = e.getNewState();
132 Logger.debug(localizationManager.getString("window.state.changed") + ": " + localizationManager.getString("old.state") + "=" + oldState + ", " + localizationManager.getString("new.state") + "=" + newState);
133 if ((newState == Frame.NORMAL) &&
134 (oldState == Frame.MAXIMIZED_BOTH || oldState == Frame.ICONIFIED)) {
135 frame.setSize(NORMAL_SIZE);
136 centerWindow(frame);
137 SwingUtilities.invokeLater(() -> saveWindowState(frame));
138 }
139 });
140
141 if (!configFile.exists()) {
142 frame.setExtendedState(Frame.MAXIMIZED_BOTH);
143 Logger.debug(localizationManager.getString("first.launch.maximized"));
144 return;
145 }
146
147 try {
148 int state = Integer.parseInt(props.getProperty("main.state", String.valueOf(Frame.MAXIMIZED_BOTH)));
149 Logger.debug(localizationManager.getString("loaded.window.state") + ": state=" + state);
150 frame.setExtendedState(state);
151
152 if (state == Frame.NORMAL) {
153 int x = Integer.parseInt(props.getProperty("main.x", "-1"));
154 int y = Integer.parseInt(props.getProperty("main.y", "-1"));
155 int width = Integer.parseInt(props.getProperty("main.width", String.valueOf(NORMAL_SIZE.width)));
156 int height = Integer.parseInt(props.getProperty("main.height", String.valueOf(NORMAL_SIZE.height)));
157
158 width = Math.max(width, NORMAL_SIZE.width);
159 height = Math.max(height, NORMAL_SIZE.height);
160
161 Dimension screenSize = Toolkit.getDefaultToolkit().getScreenSize();
162 if (x < 0 || y < 0 || x + width > screenSize.width || y + height > screenSize.height) {
163 Logger.debug(localizationManager.getString("invalid.coordinates") + " (x=" + x + ", y=" + y + "), " + localizationManager.getString("center.window"));
164 centerWindow(frame);
165 } else {
166 frame.setBounds(x, y, width, height);
167 Logger.debug(localizationManager.getString("set.saved.coordinates") + ": x=" + x + ", y=" + y + ", width=" + width + ", height=" + height);
168 }
169 }
170
171 for (JInternalFrame internalFrame : desktopPane.getAllFrames()) {
172 String translationKey = (String) internalFrame.getClientProperty("translationKey");
173 if (translationKey != null) {
174 loadInternalFrameState(internalFrame, translationKey, props);
175 }
176 }
177 } catch (NumberFormatException e) {
178 Logger.error(localizationManager.getString("window.state.load.error") + ": " + e.getMessage());
179 frame.setExtendedState(Frame.MAXIMIZED_BOTH);
180 Logger.debug(localizationManager.getString("load.error.maximized"));
181 }
182 }
